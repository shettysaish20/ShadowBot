class DAG:
    def __init__(self):
        """
        Initialize an empty DAG.

        The DAG is represented using a dictionary of nodes,
        where each node is an instance of the DAGNode class.
        """
        self.nodes = {}

    def add_node(self, node_name):
        """
        Add a new node to the DAG.

        If the node already exists, a message is printed.

        Parameters:
        - node_name (str): The name of the node to add.
        """
        if node_name not in self.nodes:
            self.nodes[node_name] = DAGNode(node_name)
        else:
            print(f"Node {node_name} already exists.")

    def add_edge(self, parent_name, child_name):
        """
        Add a directed edge from parent to child.

        If either the parent or child node does not exist, an error message is printed.

        Parameters:
        - parent_name (str): The name of the parent node.
        - child_name (str): The name of the child node.
        """
        if parent_name not in self.nodes or child_name not in self.nodes:
            print("One or both nodes do not exist.  Please add the nodes first before adding the edge.")
            return

        parent_node = self.nodes[parent_name]
        child_node = self.nodes[child_name]

        if child_node not in parent_node.children:
            parent_node.children.append(child_node)
        if parent_node not in child_node.parents:
            child_node.parents.append(parent_node)

    def has_cycle_util(self, node_name, visited, stack):
        """
        Utility function to check for cycles using DFS (Depth-First Search).

        Parameters:
        - node_name (str): The name of the current node being visited.
        - visited (dict): A dictionary to track visited nodes.
        - stack (dict): A dictionary to track nodes currently in the recursion stack.

        Returns:
        - bool: True if a cycle is found, False otherwise.
        """
        visited[node_name] = True
        stack[node_name] = True

        for child_node in self.nodes[node_name].children:
            child_name = child_node.name
            if not visited[child_name]:
                if self.has_cycle_util(child_name, visited, stack):
                    return True
            elif stack[child_name]:
                return True

        stack[node_name] = False
        return False

    def has_cycle(self):
        """
        Check if the DAG contains any cycles.

        This function uses the has_cycle_util function to perform a DFS on the graph.

        Returns:
        - bool: True if the DAG contains a cycle, False otherwise.
        """
        visited = {node_name: False for node_name in self.nodes}
        stack = {node_name: False for node_name in self.nodes}

        for node_name in self.nodes:
            if not visited[node_name]:
                if self.has_cycle_util(node_name, visited, stack):
                    return True
        return False

    def topological_sort_util(self, node_name, visited, stack):
        """
        Utility function for topological sorting using DFS.

        Parameters:
        - node_name (str): The name of the current node being visited.
        - visited (dict): A dictionary to track visited nodes.
        - stack (list): A list to store the topological order of nodes.
        """
        visited[node_name] = True

        for child_node in self.nodes[node_name].children:
            child_name = child_node.name
            if not visited[child_name]:
                self.topological_sort_util(child_name, visited, stack)

        stack.insert(0, node_name)

    def topological_sort(self):
        """
        Perform topological sorting on the DAG.

        This function uses the topological_sort_util function to perform a DFS on the graph.

        Returns:
        - list: A list of nodes in topological order.
        """
        visited = {node_name: False for node_name in self.nodes}
        stack = []

        for node_name in self.nodes:
            if not visited[node_name]:
                self.topological_sort_util(child_name, visited, stack)

        return stack

Bash Script for Disk Space Monitoring

This bash script automates the monitoring of disk space. It checks the disk usage and sends an alert if the usage exceeds a certain threshold (e.g., 80%).

#!/bin/bash

# Set the threshold for disk usage percentage
THRESHOLD=80

# Get the current disk usage percentage
USAGE=$(df -h / | awk \'NR==2{print $5}\' | tr -d \'%')

# Check if the disk usage exceeds the threshold
if [ $USAGE -gt $THRESHOLD ]; then
    echo "Disk usage is above ${THRESHOLD}%: ${USAGE}%" | mail -s "Disk Space Alert" your_email@example.com
else
    echo "Disk usage is normal: ${USAGE}%"
fi
